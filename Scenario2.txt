void main() {
  double accountBalance = 1000.0;
  //
  // 1 = Check, 2 = Deposite, 3 = Withdraw
  
  int userChoice = 2; // This is the example of user choice
  
  if(userChoice == 1){
    print("Your account balance : ${accountBalance.toStringAsFixed(2)}");
  }else if (userChoice == 2) {
    deposit(accountBalance, 1500.0); //The number 1500.0 is example user deposit
  }else if (userChoice == 3){
    withDraw(accountBalance, 500); //The 500 is example of user withdraw
  }
}

double deposit(double currentBalance, double amount) {
  if (amount > 0) {
    double newBalance = currentBalance + amount;
    print("Deposite : ${amount.toStringAsFixed(2)}");
    print("New Balance : ${newBalance.toStringAsFixed(2)}");
    return newBalance;
  }else {
    print("Invalid deposit amount. Must be greater than 0.");
    return currentBalance;
  }
}

double withDraw(double currentBalance, double amount) {
  if (amount > 0) {
    double newBalance = currentBalance - amount;
    print("Withdraw : ${amount.toStringAsFixed(2)}");
    print("New Balance : ${newBalance.toStringAsFixed(2)}");
    return newBalance;
  }else {
    print("Invalid withdraw amount. Must be greater than 0.");
    return currentBalance;
  }
}
